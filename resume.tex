% FORMATTING MACRO {{{

% Format section headings
\titleformat{\section}{\bfseries\Large}{}{0pt}{}[\vspace{1pt}\titlerule\vspace{-6.5pt}]

% Format bullet points
\renewcommand\labelitemi{$\vcenter{\hbox{\small$\bullet$}}$}
\setlist[itemize]{itemsep=-2pt, leftmargin=12pt}

% Format Education entry: Degree
% Usage: \Degree{<1-date>}{<2-school>}{<3-major>}{<4-location>}{<5-url>}{<6-info}
\newcommand*{\Degree}[6]{
  \textbf{\href{#5}{#2}} \hfill {\footnotesize #4} \\
  {#3} \hfill {\small #1} \\
  {\footnotesize #6}
}

% Format Experience entry: Job 
% Usage: \Job{<1-date>}{<2-title>}{<3-company>}{<4-location>}{<5-url}{<6-description}
\newcommand*{\Job}[6]{
  \textbf{#2} | \href{#5}{#3} --- {\small #4} \hfill {\small #1} \\
  \vspace{-9pt}
  \begin{itemize}
    {#6}
  \end{itemize}
}

% Format Projects entry: Project
% Usage: \Project{<1-title>}{2-url-name}{<3-url-link>}{<4-description>}
\newcommand*{\Project}[4]{
  \textbf{#1} \hfill \href{#3}{#2} \\
  \vspace{-9pt}
  \begin{itemize}
    {#4}
  \end{itemize}
}

% Format Projects entry: ProjectItem
% Usage: \ProjectItem{<1-title>}{2-url}{<3-description}
\newcommand*{\ProjectItem}[3]{
  \item \textbf{\href{#2}{#1}:} {#3}
}
